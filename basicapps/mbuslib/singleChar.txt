FUNCTION singleCharTx (itf$)
	IF itf$="WMbus" THEN
		WMbusWrite("E5")
	ELSE IF itf$="Mbus"
		MbusWrite("E5")
	END IF
END FUNCTION

FUNCTION singleCharRx (itf$)
	LOCAL rsp$
	'waiting for 11 bit time and (330 bit time + 50ms) before responding
	pause((11+330)/baud+50)
	
	IF itf$="WMbus" THEN
		rsp$=WMbusRead()
	ELSE IF itf$="Mbus"
		rsp$=MbusRead("")
	END IF
END FUNCTION
