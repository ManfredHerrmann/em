' frameType$	the type of the frame (single character, short, long or control frame)

FUNCTION calChecksum (frameType$,C%,A%,CI%,userData$)
	LOCAL checksum%
	
	'The checksum value depends on the frame type
	IF frameType$="shortFrame" THEN
		checksum%=C%
		checksum%+=A%
		
	ELSE IF frameType$="controlFrame" OR frameType$="longFrame" THEN
		checksum%=C%
		checksum+=A%
		checksum+=CI%
	END IF
	
	calChecksum=checksum%
END FUNCTION
